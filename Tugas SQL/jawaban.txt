create table users (
    -> id int primary key auto_increment,
    -> name varchar(255),
    -> email varchar(255),
    -> password varchar(255));
create table items (
    -> id int auto_increment primary key,
    -> name varchar(255),
    -> description varchar(255),
    -> price int,
    -> stock int,
    -> category_id int, foreign key (category_id) references categories (id));
create table categories (
    -> id int auto_increment primary key,
    -> name varchar(255));
